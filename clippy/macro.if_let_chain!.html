Produce a nested chain of if-lets and ifs from the patterns:

if_let_chain! {
    [
        let Some(y) = x,
        y.len() == 2,
        let Some(z) = y,
    ],
    {
        block
    }
}

if let Some(y) = x {
    if y.len() == 2 {
        if let Some(z) = y {
            block
        }
    }
